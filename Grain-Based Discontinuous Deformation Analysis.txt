	/* l-j=i2-i1 => l=i2 l+1=i2+1 j=i1     j l+1 same */ 
	w2 = (d[2][1] - d[1][1]) * (d[2][1] - d[1][1])
		+ (d[2][2] - d[1][2]) * (d[2][2] - d[1][2]);

		for (j = i1; j <= i2; j++)
		{
			for (l = i1; l <= i2; l++)

				if (l + 1 == j) goto a101;
				if (l - j == i2 - i1) goto a101;
				x1 = d[j][1];
				yi = d[j][2];
				x2 = d[l][1];
				y2 = d[l][2];
				x3 = d[l + 1][1];
				y3 = d[l + 1][2];
				dist();
				if (w2 > d5)  w2 = d5;   
			a101:;
			}  /*  l  */
		}  /*  j  */
	}  /*  i  */
	x3 = w2;
	fprintf(fl0, "###### minimum v-e distance ###### %lf \n", w2);
	/**************************************************/
	/* compute minimum block angle                    */
	angl();
	w4 = 180;
	for (i = 1; i <= n1; i++)
	{
		i1 = k0[i][1];
		i2 = k0[i][2];
		for (j = i1; j <= i2; j++)
		{
			d5 = u[j][2];
			if (w4 > d5)  w4 = d5;   
		}  /*  j  */
	}  /*  i  */
	fprintf(fl0, "###### minimum block  angle ###### %lf \n", w4);
	/*------------------------------------------------*/
}

			if (d[j][0] < 1)  d[j][0] = 1;
			if (d[j][0] > nj)  d[j][0] = nj;  
		}  /*  j  */
	}  /*  i  */
	/*------------------------------------------------*/
	/* movements at fixed load measured points        */
	for (i = 1; i <= np; i++)
	{
		for (j = 1; j <= 3; j++)
		{
			c[i][j] = 0; 
		}  /*  j  */
	}  /*  i  */
	/*------------------------------------------------*/
	/* initial zero for solution in begining          */
	for (i = 1; i <= n1; i++)
	{
		for (j = 1; j <= 6; j++)
		{
			z[i][j] = 0;   
		}  /*  j  */
	}  /*  i  */
	/*------------------------------------------------*/
	cons();
}

/**************************************************/
/* df04: contact finding by distance criteria     */
/**************************************************/
void df04()
{
	double sign();
	if (k6 == 1)  fprintf(fl0, "--- 4000--- \n");
	/*------------------------------------------------*/
	/* save contact positions of previous step        */
	if (nn == 1) goto a401;  
	for (i = 1; i <= n2; i++)
	{
		for (j = 0; j <= 6; j++)
		{
			m1[i][j] = m[i][j];
		}  /*  j  */
	}  /*  i  */    
	/*------------------------------------------------*/
	/* save o2 into o5 and correct o3 o4 for transfer */
	for (i = 1; i <= n2; i++)
	{
		o[i][5] = o[i][2]; 
		if (m0[i][2] == 1 && m[i][0] == 0) o[i][4] = .0000000000001 * w0 * sign(o[i][4]); 
		if (m0[i][2] == 0) o[i][3] = 0;   
		if (m0[i][2] == 0 || m[i][0] == 1) o[i][4] = 0;   
		m0[i][3] = m0[i][2]; 
	}  /*  i  */
	/*------------------------------------------------*/
	/* releas tension or shear locks                  */
	for (i = 1; i <= n2; i++)
	{
		if (m0[i][2] != 2)
		{
			m0[i][0] = 0;
		}
		if (m0[i][0] != 2)
		{
			m0[i][0] = 0;
		}
		
	}  /*  i  */
	n6 = n2;     
a401:;
	/**************************************************/
	/* if d0 small distance is main contact creteria  */
	/* c0[][] xl xu yl yu                             */
	for (i = 1; i <= n1; i++)
	{
		i1 = k0[i][1];  
		i2 = k0[i][2]; 
		c0[i][1] = d[i1][1]; 
		c0[i][2] = d[i1][1];  
		c0[i][3] = d[i1][2]; 
		c0[i][4] = d[i1][2];  
		for (j = i1; j <= i2; j++)
		{
			if (c0[i][1] > d[j][1])  c0[i][1] = d[j][1]; 
			if (c0[i][2] < d[j][1])  c0[i][2] = d[j][1]; 
			if (c0[i][3] > d[j][2])  c0[i][3] = d[j][2];
			if (c0[i][4] < d[j][2])  c0[i][4] = d[j][2]; 
		}  /*  j  */
	}  /*  i  */
	/**************************************************/
	/* contact finding by distance criteria           */
	/* d0 : normal external    distance for contact   */
	/* d0 : normal penetration distance for contact   */
	/* m6 : number of contact vertices                */
	m6 = 0;                                         
	/*------------------------------------------------*/
	/* if two boxes of blocks are near                */
	for (ii = 1; ii <= n1 - 1; ii++)
	{
		for (jj = ii + 1; jj <= n1; jj++)
		{
			if (c0[ii][2] + d0 < c0[jj][1]) goto a407;   
			if (c0[jj][2] + d0 < c0[ii][1]) goto a407;   
			if (c0[ii][4] + d0 < c0[jj][3]) goto a407;   
			if (c0[jj][4] + d0 < c0[ii][3]) goto a407;  
			/*------------------------------------------------*/
			for (i = k0[ii][1]; i <= k0[ii][2]; i++)
			{
				for (j = k0[jj][1]; j <= k0[jj][2]; j++)
				{
					x1 = d[i][1];
					yi = d[i][2];  
					x2 = d[i + 1][1];
					y2 = d[i + 1][2];  
					x3 = d[j][1];
					y3 = d[j][2];  
					x4 = d[j + 1][1];
					y4 = d[j + 1][2];  
					/*------------------------------------------------*/    
					/* vi-vj contact                       vi concave */
					/* 2 v-e contact v_j-e_i+1 e_i       v_j-e_ie_i-1 */
					if ((x1 - x3) * (x1 - x3) + (yi - y3) * (yi - y3) > d0 * d0) goto a404;  
					if (u[i][2] < 180 + .0001) goto a402;   

					m6 += 1;              
					k[m6] = (-1) * j;    
					k1[m6] = i;                           
					/*------------------------------------------------*/
					m6 += 1;              
					k[m6] = (-1) * j;    
					k1[m6] = i - 1;    

					if (i == k0[ii][1])  k1[m6] = k0[ii][2];  
					goto a406;
					/*------------------------------------------------*/
					/* vi-vj contact                       vj concave */
					/* 2 v-e contact v_i-e_j+1 e_j       v_i-e_j e_i-j */    //v_i-e_j e_j-1
				a402:;
					if (u[j][2] < 180 + .0001) goto a403;
					m6 += 1;              
					k[m6] = (-1) * i;    
					k1[m6] = j;    
					/*------------------------------------------------*/
					m6 += 1;              
					k[m6] = (-1) * i;   
					k1[m6] = j - 1;  
					if (j == k0[jj][1])  k1[m6] = k0[jj][2];
					goto a406;
					/*------------------------------------------------*/
					/* vi-vj contact                angle vi vj <=180 */  
				a403:;
					m6 += 1;        
					k[m6] = i;   
					k1[m6] = j;    
					goto a406;
					/**************************************************/
					/* (p1 p2): inner  normal of line p1p2            */
					/* (p3 p4): vector along     line p1p2            */
				a404:;
					x5 = (x1 + x2) * .5;
					y5 = (yi + y2) * .5;  
					a1 = sqrt((x2 - x1) * (x2 - x1) + (y2 - yi) * (y2 - yi));     
					p1 = (yi - y2) / a1;
					p2 = (x2 - x1) / a1;   
					p3 = (x2 - x1) / a1;
					p4 = (y2 - yi) / a1;   
					/*------------------------------------------------*/
					/* (p5 p6): inner  normal of line p3p4            */
					/* (p7 p8): vector along     line p3p4            */
					x6 = (x3 + x4) * .5;
					y6 = (y3 + y4) * .5;  
					a2 = sqrt((x4 - x3) * (x4 - x3) + (y4 - y3) * (y4 - y3));    
					p5 = (y3 - y4) / a2;
					p6 = (x4 - x3) / a2;   
					p7 = (x4 - x3) / a2;
					p8 = (y4 - y3) / a2;   
					/*------------------------------------------------*/
					d1 = p5 * (x1 - x6) + p6 * (yi - y6);  
					d2 = p7 * (x1 - x6) + p8 * (yi - y6);    
					d3 = p1 * (x3 - x5) + p2 * (y3 - y5);    
					d4 = p3 * (x3 - x5) + p4 * (y3 - y5);    
					/*------------------------------------------------*/
					/* vi - ej+1 j contact       h5=0.4  d0 in dstn() */
					if ((x1 - x4) * (x1 - x4) + (yi - y4) * (yi - y4) <= d0 * d0) goto a405; 
					if (fabs(d2) > a2 / 2) goto a405;   
					if (d1<-d0 || d1>h5 * d0) goto a405;  
					m6 += 1;             
					k[m6] = (-1) * i;     
					k1[m6] = j;     
					/*------------------------------------------------*/
					/* vj - ei+1 i contact                            */
				a405:;
					if ((x3 - x2) * (x3 - x2) + (y3 - y2) * (y3 - y2) <= d0 * d0) goto a406;  
					if (fabs(d4) > a1 / 2) goto a406;  
					if (d3<-d0 || d3>h5 * d0) goto a406;   
					m6 += 1;               
					k[m6] = (-1) * j;    
					k1[m6] = i;     
					/*------------------------------------------------*/
				a406:;
				}  /*  j  */
			}  /*  i  */
		a407:;
		}  /*  jj */
	}  /*  ii */

}


/**************************************************/
/* df05: contact finding by angle criteria        */
/**************************************************/
void df05()
{
	if (k6 == 1)  fprintf(fl0, "--- 5000--- \n");
	/*------------------------------------------------*/
	/* n2 number of contacts m[]                      */
	n2 = 0;
	for (i = 1; i <= m6; i++)
	{
		/*------------------------------------------------*/
		/* vector i2i1 i2i3  j2j1 j2j3 rotate from x to y */
		/* in block i3 i2 i1  j3 j2 j1 rotate from x to y */
		i2 = labs(k[i]);  
		j2 = k1[i];  
		i0 = n0[i2];      
		jo = n0[j2];     
		i3 = i2 - 1;   
		if (i2 == k0[i0][1])  i3 = k0[i0][2];  
		j3 = j2 - 1;   
		if (j2 == k0[jo][1])  j3 = k0[jo][2];   
		i1 = i2 + 1;   
		if (i2 == k0[i0][2])  i1 = k0[i0][1];   
		ji = j2 + 1;   
		if (j2 == k0[jo][2])  ji = k0[jo][1];   
		/*------------------------------------------------*/
		/* direction angle of i2i1 i2i3 j2j1 j2j3  ox=>oy */
		e[1][1] = u[i2][1];      
		e[2][1] = u[i3][1] - 180;   
		e[3][1] = u[j2][1];       
		e[4][1] = u[j3][1] - 180;   
		/*------------------------------------------------*/
		/* v - e 180 angle j j2j1 j1j2 rotate from x to y */
		if (k[i] < 0)  e[4][1] = e[3][1] - 180;   
		for (j = 1; j <= 4; j++)
		{
			if (e[j][1] < 0)  e[j][1] += 360;
		}  /*  j  */
		/*------------------------------------------------*/
		/* e1 angle i2 i2i1=>i2i3  e2 angle j2 j2j1=>j2j3 */
		e1 = e[2][1] - e[1][1];
		e2 = e[4][1] - e[3][1];
		if (e1 < 0)  e1 += 360;  
		if (e2 < 0)  e2 += 360;   
		/*------------------------------------------------*/
		/* e3 angle i2i1=>j2j1        e4 angle i2i3=>j2j3 */
		e3 = e[3][1] - e[1][1];
		e4 = e[4][1] - e[2][1];
		if (e3 < 0)  e3 += 360;   
		if (e4 < 0)  e4 += 360;   
		/*------------------------------------------------*/
		/* e5 j2j3=>i2i1  e6 i2i3=>j2j1      e5 e6 may <0 */  
		e5 = 360 - e3 - e2;    
		e6 = e3 - e1;       
		/*------------------------------------------------*/
		/* 1st angle e1=>e2 > 2nd angle e3=>e4  for check */
		/* e[0][2] check if whole 2nd angle lay in 1st    */
		e[1][2] = e[1][1];
		e[2][2] = e[1][1] + e1;
		e[3][2] = e[3][1];
		e[4][2] = e[4][1];
		e[0][2] = e[3][1] + 0.5 * e2;
		if (e1 > e2) goto a501;   
		e[1][2] = e[3][1];     
		e[2][2] = e[3][1] + e2;
		e[3][2] = e[1][1];
		e[4][2] = e[2][1];
		e[0][2] = e[1][1] + 0.5 * e1;  
		/*------------------------------------------------*/
		/* angle overlay check for contacts               */
		/* h1 angle overlay allowance=3 degree     cons() */
	a501:;
		if (e[1][2] + h1 < e[0][2] && e[0][2] < e[2][2] - h1) goto a508;  
		if (e[1][2] + h1 < e[3][2] && e[3][2] < e[2][2] - h1) goto a508;  
		if (e[1][2] + h1 < e[4][2] && e[4][2] < e[2][2] - h1) goto a508;  
		if (e[1][2] + h1 < e[0][2] + 360 && e[0][2] + 360 < e[2][2] - h1) goto a508;  
		if (e[1][2] + h1 < e[3][2] + 360 && e[3][2] + 360 < e[2][2] - h1) goto a508;  
		if (e[1][2] + h1 < e[4][2] + 360 && e[4][2] + 360 < e[2][2] - h1) goto a508;  
		if (k[i] > 0) goto a502;   
		/*------------------------------------------------*/
		/* v-e v=i2 edge j1j2   m[][0]=0 v-e m[][0]=1 v-v */
		/* m[][4] m[][5] 2 side of i2 for lock legth      */
		n2 += 1;
		m[n2][0] = 0;  
		m[n2][1] = i2; 
		m[n2][2] = ji;
		m[n2][3] = j2;  
		m[n2][4] = 0;
		m[n2][5] = 0;
		m[n2][6] = 0;
		if (e5 < h1)  m[n2][4] = i1;  
		if (e6 < h1)  m[n2][5] = i3;  
		goto a508;
		/**************************************************/
		/* v-v vertex i2 vertex j2        i2<=180 j2<=180 */
		/* reduce to v-e when adjacent lines are parallel */
	a502:;
		if (e5 < h1) goto a503; 
		if (e6 < h1) goto a504; 
		goto a505;
		/*------------------------------------------------*/
		/* contact edge of e5 as entrance        m[][6]>0 */
		/* both e5<h1 e6<h1  both angle i2 j2 =180+-2h    */
	a503:;
		n2 += 1;
		m[n2][0] = 0;   
		m[n2][1] = i2;  
		m[n2][2] = j2;
		m[n2][3] = j3;  
		m[n2][4] = j2;  
		m[n2][5] = i1;
		m[n2][6] = i2;  
		if (e6 < h1) goto a504;
		goto a508;
		/*------------------------------------------------*/
		/* contact edge of e6 as entrance        m[][6]>0 */
	a504:;
		n2 += 1;
		m[n2][0] = 0;   
		m[n2][1] = j2;  
		m[n2][2] = i2;
		m[n2][3] = i3;  
		m[n2][4] = i2;  
		m[n2][5] = ji;
		m[n2][6] = j2;  
		goto a508;
		/**************************************************/
		/* v-v i2<180 j2<180   entrance line j1j2 or i1i2 */
		/* entrance symmetry if e3=180+-.3h1 take large y */
	a505:;
		a1 = 0.5 * e1 + e[1][1];    
		a2 = 0.5 * e2 + e[3][1];  
		d1 = fabs(sin(dd * a1));
		d2 = fabs(sin(dd * a2));
		n2 += 1;
		m[n2][0] = 1;   
		m[n2][1] = i2;  
		m[n2][2] = ji;
		m[n2][3] = j2;  
		b1 = e[3][1]; 
		if (e3 <= 180 - .3 * h1 || (e3 <= 180 + .3 * h1 && d2 >= d1)) goto a506;
		m[n2][1] = j2; 
		m[n2][2] = i1;
		m[n2][3] = i2;  
		b1 = e[1][1];   
		/*------------------------------------------------*/
		/* v-v i2<180 j2<180   entrance line i2i3 or j2j3 */
		/* entrance symmetry if e4=180+-.3h1 take large y */
	a506:;
		m[n2][4] = j2;  
		m[n2][5] = i2;
		m[n2][6] = i3; 
		b2 = e[2][1];   
		if (e4 <= 180 - .3 * h1 || (e4 <= 180 + .3 * h1 && d1 >= d2)) goto a507;
		m[n2][4] = i2; 
		m[n2][5] = j2;
		m[n2][6] = j3;  
		b2 = e[4][1];   
		/*------------------------------------------------*/
		/* make 1st entrance symmetry     initialy locked */
	a507:;
		d1 = fabs(sin(dd * b1));
		d2 = fabs(sin(dd * b2));
		if (d1 >= d2) goto a508;  
		for (j = 1; j <= 3; j++)
		{
			j6 = m[n2][j];
			m[n2][j] = m[n2][j + 3];
			m[n2][j + 3] = j6;    
		}  /*  j  */
	a508:;
		if (n2 == 11 * n1 - 1) fprintf(fl0, "@@@ i= %d n2=%d \n", i, n2);
		/* if (n2==11*n1-1) fprintf(fl0,"@@@m6= i=%d  n2=%d \n",i,n2); */
	}  /*  i  */
	/*------------------------------------------------*/
}

/**************************************************/
/* df06: contact transfer                         */
/**************************************************/
void df06()
{
	double sign();
	if (k6 == 1)  fprintf(fl0, "--- 6000--- \n");
	/*------------------------------------------------*/
	/* initiate o0 o1 o2 of next step for transfer    */
	/* k[] flag of contact point transfer df07 proj() */
	for (i = 1; i <= n2; i++)
	{
		m0[i][1] = 0;   
		m0[i][2] = 0;   
		o[i][0] = 0;   
		o[i][1] = 0;   
		o[i][2] = 0; 
		o[i][6] = 0;  
		k[i] = 0;   
	}  /*  i  */
	/**************************************************/
	/* mm[][0-1] start-end   first block index of m[] */
	for (i = 1; i <= n1; i++)
	{
		mm[i][0] = 0;  
	}  /*  i  */
	/*------------------------------------------------*/
	for (i = 1; i <= n2; i++)
	{
		i1 = m[i][1];   
		ji = m[i][2];     
		i2 = n0[i1];      
		j2 = n0[ji];     
		if (i2 > j2)  i2 = j2;  
		mm[i2][0] += 1;     
	}  /*  i  */
	/*------------------------------------------------*/
	mm[0][1] = 0;
	for (i = 1; i <= n1; i++)
	{
		mm[i][1] = mm[i][0] + mm[i - 1][1]; 
		mm[i][0] = mm[i][1] - mm[i][0] + 1; 
	}  /*  i  */
	/**************************************************/
	/* m0[][0] tension flag save to k1[] to transfer  */  
	if (nn == 1) goto a606;
	for (i = 1; i <= n6; i++) 
	{
		k1[i] = m0[i][0];    
	}  /*  i  */
	/*------------------------------------------------*/
	for (i = 1; i <= n2; i++) 
	{
		m0[i][0] = 0;  
	}  /*  i  */
	/**************************************************/
	/* transfer contacts  m1[6]>0 for co-edge   nn>=1 */
	for (ii = 1; ii <= n6; ii++) 
	{
		m0[ii][4] = 0; 
		if (m0[ii][3] == 0) goto a602;
		if (m0[ii][3] == 3) m0[ii][4] = 1; 
		if (m0[ii][3] == 3) m0[ii][3] = 0; 
		/*------------------------------------------------*/
		i0 = m1[ii][0]; 
		/* if (m1[ii][6] > 0) i0=1;                       */
		for (jj = 0; jj <= i0; jj++)
		{
			i1 = m1[ii][1];  
			ji = m1[ii][2];  
			i2 = n0[i1];   
			j2 = n0[ji];     
			if (i2 > j2)  i2 = j2;  
			/*------------------------------------------------*/
			/* i2 same start block         m[6]>0 for co-edge */
			for (i = mm[i2][0]; i <= mm[i2][1]; i++)  
			{
				jo = m[i][0];  
				if (m[i][6] > 0) jo = 1;
				for (j = 0; j <= jo; j++)
				{
					if (m0[ii][jj + 3] == 0) goto a601;  
					if (m[i][3 * j + 1] != m1[ii][3 * jj + 1]) goto a601;
					if (m[i][3 * j + 2] != m1[ii][3 * jj + 2]) goto a601;
					if (m[i][3 * j + 3] != m1[ii][3 * jj + 3]) goto a601; 
					/*------------------------------------------------*/
					/* last step o3 o4 o5 goto next step o0 o1 o2     */
					/* m0=3 o3 for other spring  o1 other normal dspl */
					o[i][0] = o[ii][3];  
					o[i][1] = o[ii][4]; 
					o[i][2] = o[ii][5]; 
					o[i][6] = o[ii][7];  
					if (j == 0 && m[i][0] == 1) o[i][0] = o[ii][3]; 
					if (j == 0 && m[i][0] == 1) o[i][1] = 0;         
					if (j == 1 && m[i][0] == 1) o[i][0] = 0;          
					if (j == 1 && m[i][0] == 1) o[i][1] = o[ii][3];   
					/*------------------------------------------------*/
					/* k[] flag of contact point transfer df07 proj() */
					/* 2 v-e j=0 jj=0 same edge k[]=1 keep lock point */
					/* m0[0-2] transfer v-v transfer v-e set m0[][]=2 */
					m0[i][j + 1] = m0[ii][jj + 3]; 
					m0[i][0] = k1[ii];      
					if (m[i][0] == 0 && m0[ii][3] == 2 && jj == 0) k[i] = 1;   
					if (m[i][0] == 0 && m1[ii][0] > 0)  m0[i][j + 1] = 2;    
				a601:;                                             
				}  /*  j  */                                    
			}  /*  i  */
		}  /*  jj */
	a602:;
	}  /*  ii */
	/**************************************************/
	/* m0[][1]=0  m0[][2]=1 => m0[][2]=3   v-e to v-v */
	/* m[][0]=0 i2>0 co-edge match m[][4-6]=>m[][1-3] */
	/* set m0[i][1]=0    before open-close iterations */
	for (i = 1; i <= n2; i++)
	{
		if (m[i][0] == 0)  goto a603; 
		i1 = m0[i][1];
		i2 = m0[i][2];
		if (i1 != 0)  m0[i][2] = 1; 
		if (i2 != 0)  m0[i][2] = 3; 
		m0[i][1] = 0; 
	a603:;
	}  /*  i  */
	/*------------------------------------------------*/
	/* to v-e one transfer line                       */
	for (i = 1; i <= n2; i++)
	{
		if (m[i][0] == 0 && m[i][6] == 0) m0[i][2] = m0[i][1];  
		if (m[i][0] == 0 && m[i][6] == 0) m0[i][1] = 0;  
	}  /*  i  */
	/**************************************************/
	/* v-v to v-e co-edge     set i1 in between i2 i3 */
	/* switch m[][j] m[][j+3] o[][2] cange o0 o1 same */
	/* m0[][1-2]=2 keep old penetration & line point  */
	for (i = 1; i <= n2; i++)
	{
		if (m[i][0] == 1 || m[i][6] == 0 || m0[i][1] == 2) goto a604;
		i1 = m[i][1]; 
		i2 = m[i][2];
		i3 = m[i][3];
		x1 = d[i2][1] - d[i1][1];
		yi = d[i2][2] - d[i1][2];  
		x2 = d[i3][1] - d[i1][1];
		y2 = d[i3][2] - d[i1][2];  
		d1 = (x1 * x2 + yi * y2); 
		if (d1 <= 0 && m0[i][2] != 2) goto a604; 
		/*------------------------------------------------*/
		m[i][1] = m[i][4];
		m[i][2] = m[i][5];
		m[i][3] = m[i][6];
		m[i][4] = i1;
		m[i][5] = i2;
		m[i][6] = i3;
	a604:;
	}  /*  i  */
	/*------------------------------------------------*/
	/* to v-e two transfer lines                      */
	for (i = 1; i <= n2; i++)
	{
		if (m[i][0] == 1 || m[i][6] == 0) goto a605;
		if (m0[i][2] == 0)  m0[i][2] = m0[i][1]; 
		m0[i][1] = 0; 
	a605:;
	}  /*  i  */
	/**************************************************/
	/* m1[i][0] joint material number  edge m[i][3-2] */
	/* m[][2]-m[][3]=>d[i+1][]-d[i][] d[i+1][0] joint */
a606:;
	l4 = 0;  
	for (i = 1; i <= n2; i++)
	{
		if (m0[i][2] != 0) l4 += 1;
		i1 = m[i][2];   
		m1[i][0] = (long)d[i1][0];
	}  /*  i  */
	/*------------------------------------------------*/
}

/**************************************************/
/* df07: contact initialization                   */
/**************************************************/
void df07()
{
	void proj();
	if (k6 == 1)  fprintf(fl0, "--- 7000--- \n");
	/*------------------------------------------------*/
	/* o5 contact length     o1 set sliding direction */
	/* initiate o3 o4 o5 for comparation of sliding   */
	for (i = 1; i <= n2; i++)
	{
		if (m0[i][2] == 1 && m[i][0] == 0) o[i][1] = .0000000000001 * w0 * sign(o[i][1]);
		if (m0[i][2] == 1 && m[i][0] == 0) o[i][4] = o[i][1]; 
		if (m0[i][2] == 1 && m[i][0] == 0) o[i][3] = o[i][0]; 
		o[i][5] = 0;
	}  /*  i  */
	/*------------------------------------------------*/
	/* contact length and contact point     nn=1 nn>1 */
	for (i = 1; i <= n2; i++)
	{
		if (nn == 1)
		{
			m0[i][0] = 0;       
			o[i][6] = 0;
			o[i][7] = 0;
		}
		if (m[i][0] == 1) goto a705;  
		if (m[i][6] == 0) goto a701;  
		if (m[i][6] > 0) goto a704;  
		/*------------------------------------------------*/
		/* v-e => v-e m[][4]>0      set tension m0[][0]=2 */
	a701:;
		if (m[i][4] == 0) goto a702;
		i1 = m[i][1];
		i2 = m[i][2];
		i3 = m[i][3];
		i4 = m[i][4];
		if (nn == 1) m0[i][0] = 2; 
		proj();
		if (nn == 1)
		{
			o[i][7] = o[i][5] * hh1 * 2; 
			o[i][6] = o[i][5] * hh1 * 2; 
		}
		/*------------------------------------------------*/
		/* v-e => v-e m[][5]>0      set tension m0[][0]=2 */
	a702:;
		if (m[i][5] == 0) goto a703;
		i1 = m[i][1];
		i2 = m[i][2];
		i3 = m[i][3];
		i4 = m[i][5];
		if (nn == 1) m0[i][0] = 2; 
		proj();
		if (nn == 1)
		{
			o[i][7] = o[i][5] * hh1 * 2; 
			o[i][6] = o[i][5] * hh1 * 2; 
		}
		goto a705;
		/*------------------------------------------------*/
		/* v-e => v-e no co-edge      m[i][4]=0 m[i][5]=0 */
	a703:;
		i1 = m[i][1];
		i2 = m[i][2];
		i3 = m[i][3];
		i4 = 0;             
		proj();
		goto a705;
		/*------------------------------------------------*/
		/* v-v => v-e e5<h1 or e6<h1    tension m0[][0]=2 */
	a704:;
		i1 = m[i][1];
		i2 = m[i][2];
		i3 = m[i][3];
		i4 = m[i][5];
		if (nn == 1) m0[i][0] = 2; 
		proj();
		if (nn == 1)
		{
			o[i][7] = o[i][5] * hh1 * 2; 
			o[i][6] = o[i][5] * hh1 * 2; 
		}
		/*------------------------------------------------*/
		/* due transfer m[][i+3] may m[][i] m[][5]-m[][6] */   //??
		i1 = m[i][1];
		i2 = m[i][2];
		i3 = m[i][3];
		i4 = m[i][6];
		proj();
	a705:;
	}  /*  i  */
	/**************************************************/
	/* nn=1 set initial open-close state m0[][2]      */
	/* starting with m0[][2]=1 check initial contact  */
	if (nn > 1) goto a707;  
	for (i = 1; i <= n2; i++)
	{
		m0[i][2] = 1;   
		for (j = 0; j <= m[i][0]; j++)
		{
			ji = m[i][j * 3 + 1];
			j2 = m[i][j * 3 + 2];
			j3 = m[i][j * 3 + 3];
			x1 = d[ji][1];
			yi = d[ji][2];  
			x2 = d[j2][1];
			y2 = d[j2][2];  
			x3 = d[j3][1];
			y3 = d[j3][2]; 
			/*------------------------------------------------*/
			/* j1 j2 j3 rotate from x to y => no penetration  */
			/* nearly close = close       s[] normal distance */
			a1 = sqrt((x3 - x2) * (x3 - x2) + (y3 - y2) * (y3 - y2));  
			s[j + 1] = ((x2 - x1) * (y3 - yi) - (y2 - yi) * (x3 - x1)) / a1; 
			if (s[j + 1] > f0 * d0)  m0[i][2] = 0;  
		}  /*  j  */
		/*------------------------------------------------*/
		/* v-e  if close  set lock                        */
		if (m0[i][2] == 0) goto a706;   
		if (m[i][0] == 0) m0[i][2] = 2;  
		if (m[i][0] == 0) goto a706;  
		/*------------------------------------------------*/
		/* v-v  if close         choose shortest distance */
		m0[i][2] = 1;  
		if (s[1] < s[2])  m0[i][2] = 3;  
	a706:;
	}  /*  i  */
a707:;
	/*------------------------------------------------*/
}

		q[5][3] = q[3][5];

		q[6][3] = q[3][6];    
		q[4][4] = u1;

void df18()
{

	double sign();

		{
			c0[i][j] = 0;  
		}  /*  j  */
	}  /*  i  */
	/**************************************************/
	/* set off-on of normal and shear spring          */
	/* j=1 last interation         j=2 this iteration */
	/* q[j][1-2] normal-shear   q[j][3] 2nd reference */
	for (i = 1; i <= n2; i++)
	{
		for (j = 1; j <= 2; j++)
		{
			q[j][1] = 0;
			q[j][2] = 0;

			if (m0[i][j] == 2) goto b801;
			q[j][1] = 0;
			q[j][2] = 0;
			q[j][3] = 1;
		b801:;
		}  /*  j  */
		/*------------------------------------------------*/
		/* set on-off modify coefficient                  */
		/* q[1][1-2] normal-shear   q[1][3] 2nd reference */
		for (j = 1; j <= 4; j++)
		{
			q[1][j] = (q[2][j] - q[1][j]);
		}  /*  j  */
		/*------------------------------------------------*/
		/* comput 4 6*6         2 6*1 contact submatrices */
		/* sliding force +- changing shear   g0/h2 h2=2.5 */
		/* k5=0 set m0[][1]=0 still have friction force   */ //k5=1: m0[i][1]=m0[i][2]
		/* k5=0 m0[i][1]>0 goto to compute friction force */
		for (jj = 0; jj <= m[i][0]; jj++)
		{
			if (jj == 0) goto b802;
			q[1][1] = q[1][3];//jj=1
			q[1][2] = q[1][4];
		b802:;
			q[0][1] = q[1][1];
			q[0][2] = q[1][2];
			if (o[i][6] < 0.005)
			{
				q[1][1] = q[1][1] * hh;
				q[1][2] = q[1][2] * hh / h2;
			}
			else
			{
				q[1][1] = q[1][1] * o[i][6];
				q[1][2] = q[1][2] * o[i][6] / h2;
			}


			if (fabs(q[0][1]) > .5 || fabs(q[0][2]) > .5) goto b803;
			if (m[i][0] == 0 && m0[i][2] == 1 && k5 == 0) goto b803;
			if (m[i][0] == 0 && m0[i][2] == 1 && m0[i][1] != 0) goto b803;
			goto b814;
			/*------------------------------------------------*/
			/* submatrix of normal shear springs and friction */
		b803:;
			l1 = m[i][3 * jj + 1]; //1
			l2 = m[i][3 * jj + 2]; //2
			l3 = m[i][3 * jj + 3]; //3
			i1 = n0[l1];  //i
			i2 = n0[l2];  //j
			ji = k2[i1];
			j2 = k2[i2];

			x1 = d[l1][1];
			yi = d[l1][2];
			x2 = d[l2][1];
			y2 = d[l2][2];
			x3 = d[l3][1];
			y3 = d[l3][2];
			/*------------------------------------------------*/
			/* s1 normal s2 shear coefficients s3 ratio p2-p3 */
			b1 = sqrt((x3 - x2) * (x3 - x2) + (y3 - y2) * (y3 - y2)); 
			s1 = (x2 - x1) * (y3 - yi) - (y2 - yi) * (x3 - x1);
			s1 = s1 / b1;
			if (m[i][0] != 0) goto b804; 
			s3 = o[i][2];
			s2 = (x1 - (1 - s3) * x2 - s3 * x3) * (x3 - x2) + (yi - (1 - s3) * y2 - s3 * y3) * (y3 - y2);
			s2 = s2 / b1; 
		b804:;
			i0 = i1; 
			x = x1;
			y = yi;
			dspl();
			for (j = 1; j <= 6; j++)
			{
				s[j] = (y2 - y3) * t[1][j] + (x3 - x2) * t[2][j];
				s[j] /= b1; 
				if (m[i][0] != 0) goto b805; 
				s[j + 12] = (x3 - x2) * t[1][j] + (y3 - y2) * t[2][j]; 
				s[j + 12] /= b1;
			b805:;
			}  /*  j  */
			/*------------------------------------------------*/
			/* p2 terms      s07-s12 j normal s19-s24 j shear */
			/* p2 terms      s25-s32 j friction               */
			i0 = i2; //l2所在的块体号
			x = x2;
			y = y2;
			dspl();
			for (j = 1; j <= 6; j++)
			{
				s[j + 6] = (y3 - yi) * t[1][j] + (x1 - x3) * t[2][j]; 
				if (m[i][0] != 0) goto b806; 
				s[j + 18] = (-x1 + 2 * (1 - s3) * x2 - (1 - 2 * s3) * x3) * t[1][j];
				s[j + 18] += (-yi + 2 * (1 - s3) * y2 - (1 - 2 * s3) * y3) * t[2][j]; 
				s[j + 24] = (1 - s3) * ((x3 - x2) * t[1][j] + (y3 - y2) * t[2][j]); 
			b806:;
			}  /*  j  */
			/*------------------------------------------------*/
			/* p3 terms      s07-s12 j normal s19-s24 j shear */
			/* p3 terms      s25-s32 j friction               */
			i0 = i2; 
			x = x3;
			y = y3;
			dspl();
			for (j = 1; j <= 6; j++)
			{
				s[j + 6] += (yi - y2) * t[1][j] + (x2 - x1) * t[2][j];
				s[j + 6] /= b1; 
				if (m[i][0] != 0) goto b807; 
				s[j + 18] += (x1 - (1 - 2 * s3) * x2 - 2 * s3 * x3) * t[1][j];
				s[j + 18] += (yi - (1 - 2 * s3) * y2 - 2 * s3 * y3) * t[2][j];
				s[j + 18] /= b1;        
				s[j + 24] += s3 * ((x3 - x2) * t[1][j] + (y3 - y2) * t[2][j]);
				s[j + 24] /= b1; 
			b807:;
			}  /*  j  */
			/**************************************************/
			/* submatrix ii  s01-06 i normal s13-18 i shear   */
			/* 1st if for friction force                      */
			if (fabs(q[0][1]) < .5 && fabs(q[0][2]) < .5) goto b813;
			if (pp == 0)  i3 = n[ji][1] + n[ji][2] - 1; 
			if (pp == 1)  i3 = n[ji][1]; 
			for (j = 1; j <= 6; j++)
			{
				for (l = 1; l <= 6; l++)
				{
					j3 = 6 * (j - 1) + l;
					a[i3][j3] += q[1][1] * s[j] * s[l]; 
					if (fabs(q[0][2]) > .5)
						a[i3][j3] += q[1][2] * s[j + 12] * s[l + 12]; 
				}  /*  l  */
			}  /*  j  */
			/*------------------------------------------------*/
			/* submatrix jj  s07-12 j normal s19-24 j shear   */
			if (pp == 0)  i3 = n[j2][1] + n[j2][2] - 1; 
			if (pp == 1)  i3 = n[j2][1]; 
			for (j = 1; j <= 6; j++)
			{
				for (l = 1; l <= 6; l++)
				{
					j3 = 6 * (j - 1) + l;
					a[i3][j3] += q[1][1] * s[j + 6] * s[l + 6];
					if (fabs(q[0][2]) > .5) 
						a[i3][j3] += q[1][2] * s[j + 18] * s[l + 18]; 
				}  /*  l  */
			}  /*  j  */
			/*------------------------------------------------*/
			/* locate j1j2 in a[][] only lower triangle saved */
			if (pp == 0 && ji < j2) goto b809;
			if (pp == 1 && ji > j2) goto b809; 
			for (j = n[ji][1]; j <= n[ji][1] + n[ji][2] - 1; j++)
			{
				i3 = j;
				if (k[j] == j2) goto b808;   
			}  /*  j  */
			/*------------------------------------------------*/
			/* submatrix ij  s01-06 i normal s07-12 j normal   */
		b808:;
			for (j = 1; j <= 6; j++)
			{
				for (l = 1; l <= 6; l++)
				{
					j3 = 6 * (j - 1) + l;
					a[i3][j3] += q[1][1] * s[j] * s[l + 6];
					if (fabs(q[0][2]) > .5) 
						a[i3][j3] += q[1][2] * s[j + 12] * s[l + 18]; 
				}  /*  l  */
			}  /*  j  */
			k1[i3] = 1; 
			goto b811;
			/*------------------------------------------------*/
			/* locate j2j1 in a[][] only lower triangle saved */
		b809:;
			for (j = n[j2][1]; j <= n[j2][1] + n[j2][2] - 1; j++)
			{
				i3 = j;
				if (k[j] == ji) goto b810;   
			}  /*  j  */
			/*------------------------------------------------*/
			/* submatrix ji  s07-12 j normal s19-24 j shear   */
		b810:;
			for (j = 1; j <= 6; j++)
			{
				for (l = 1; l <= 6; l++)
				{
					j3 = 6 * (j - 1) + l;
					a[i3][j3] += q[1][1] * s[j + 6] * s[l]; 
					if (fabs(q[0][2]) > .5)
						a[i3][j3] += q[1][2] * s[j + 18] * s[l + 12]; 
				}  /*  l  */
			}  /*  j  */
			k1[i3] = 1;
			/*------------------------------------------------*/
			/* load term of normal & shear  s1 s2 coefficient */
		b811:;
			for (j = 1; j <= 6; j++)
			{
				f[ji][j] += -q[1][1] * s1 * s[j];   
				f[j2][j] += -q[1][1] * s1 * s[j + 6]; 
				if (fabs(q[0][2]) < .5) goto b812;
				f[ji][j] += -q[1][2] * s2 * s[j + 12];   
				f[j2][j] += -q[1][2] * s2 * s[j + 18];   
			b812:;
			}  /*  j  */
			/*------------------------------------------------*/
			/* sliding friction force      s4*g0 normal force */
		b813:;
			if (m[i][0] != 0 || m0[i][2] != 1) goto b814;
			if (m0[i][1] == 0 && k5 == 1) goto b814;
			s4 = o[i][3];
			if (s4 > 0)  s4 = 0;
			j4 = m1[i][0];  
			t1 = b0[j4][0];   
			t2 = b0[j4][1] * o[i][5];  
			e1 = 0;
			if (m0[i][0] == 2)  e1 = t2; 
			s4 = (fabs(s4) * g7 * tan(dd * t1) + e1) * sign(o[i][4]);
			for (j = 1; j <= 6; j++)
			{
				c0[ji][j] += -s4 * s[j + 12];  
				c0[j2][j] += s4 * s[j + 24];  
			}  /*  j  */
		b814:;

		}  /*  jj */
	}  /*  i  */
	/*------------------------------------------------*/

			  {
				  for(i1=1;i1<=6; i1++)
				  {
					  l=6*(j-1)+i1;
					  fprintf(f116,"%11.10lf ",a[i][l]);
				  }
				  fprintf(f116,"\n");
			  }
		  }

		  /*sprintf(aa,"f_mtrx%d_%d.dat",n9,nn);

			  {

				  {

/**************************************************/
/* di20: SOR interation equation solver           */
/**************************************************/
void di20()
{
	void invr();
	if (k6 == 1)  fprintf(fl0, "---20000--- \n");
	/*------------------------------------------------*/
	/* start new iteration                            */
	/* e1:solution error          e2:sum of solutions */
	i5 = 0;
c001:;
	e1 = 0;
	e2 = 0;
	/*------------------------------------------------*/
	/* initial value of free term                     */
	for (i = 1; i <= n1; i++)
	{
		for (j = 1; j <= 6; j++)
		{
			r[i][j] = f[i][j] + c0[i][j]; 
		}  /*  j  */
	}  /*  i  */
	/**************************************************/
	/*  inverse of diagnal 6*6 submatrices            */
	for (i = 1; i <= n1; i++)
	{
		i1 = n[i][1];
		ji = 0;
		for (j = 1; j <= 6; j++)
		{
			for (l = 1; l <= 6; l++)
			{
				ji++;
				q[j][l] = a[i1][ji]; 
			}  /*  l  */
		}  /*  j  */
		invr();
		/*------------------------------------------------*/
		ji = 0;
		for (j = 1; j <= 6; j++)
		{
			for (l = 1; l <= 6; l++)
			{
				ji++;
				v[i][ji] = e[j][l];
			}  /*  l  */
		}  /*  j  */
	}  /*  i  */    
	/**************************************************/
	/* operation of upper triangle                    */
	for (i = 1; i <= n1; i++)
	{
		for (j = n[i][1] + 1; j <= n[i][1] + n[i][2] - 1; j++)
		{
			if (k1[j] == 0) goto c002;
			i1 = k[j];   
			i0 = 0;
			for (l = 1; l <= 6; l++)
			{
				for (l1 = 1; l1 <= 6; l1++)
				{
					i0++;
					r[i][l] += -a[j][i0] * z[i1][l1];//b-[L]Xk+1-[U]Xk
				}  /*  l1 */
			}  /*  l  */
		c002:;
		}  /*  j  */
		/*------------------------------------------------*/
		/* operation of diagnal terms                     */
		i0 = 0;
		for (j = 1; j <= 6; j++)
		{
			s[j] = 0;
			for (l = 1; l <= 6; l++)
			{
				i0++;
				s[j] += v[i][i0] * r[i][l];//[D]^-1(b-[L]Xk+1-[U]Xk)
			}  /*  l  */
		}  /*  j  */
		/*------------------------------------------------*/
		/* qq: sor factor 1.0 -- 2.0     compute solution */
		for (j = 1; j <= 6; j++)
		{
			a1 = qq * (s[j] - z[i][j]) + z[i][j];//W*{[D]^-1(b-[L]Xk+1-[U]Xk)-Xk}+Xk
			e1 += fabs(a1 - z[i][j]);
			e2 += fabs(a1);
			z[i][j] = a1;
		}  /*  j  */
		/**************************************************/
		/* operation of lower triangle                    */
		for (j = n[i][1] + 1; j <= n[i][1] + n[i][2] - 1; j++)
		{
			if (k1[j] == 0) goto c003;
			i1 = k[j];   
			for (l = 1; l <= 6; l++)
			{
				i0 = l;
				for (l1 = 1; l1 <= 6; l1++)
				{
					r[i1][l] += -a[j][i0] * z[i][l1];//-[L]Xk+1
					i0 += 6;
				}  /*  l1 */
			}  /*  l  */
		c003:;
		}  /*  j  */
	}  /*  i  */
	/**************************************************/
	/* error and branch of iteration                  */
	i5 = i5 + 1;
	e1 = e1 / (e2 + .00000000001);
	/* if (i5%20 == 0) fprintf(fl0,"it %6d %lf \n",i5,e1); */
	if (e1 < .00000001) goto c004;
	if (i5 > 200) goto c004;
	/*------------------------------------------------*/
	/* goto next iteration                            */
	goto c001;
c004:;
	fprintf(fl0, "end%6d %lf \n", i5, e1);
	/* fprintf(fl0,"end %6d \n",i5); */
	/*------------------------------------------------*/
}

	/*------------------------------------------------*/
	/* save f[]          add frinction loading to f[] */
	for (i = 1; i <= n1; i++)
	{
		for (j = 1; j <= 6; j++)
		{
			r[i][j] = f[i][j];
			f[i][j] += c0[i][j];
		}  /*  j  */
	}  /*  i  */
	/**************************************************/
	/* L U triangle   A=L D^(-1) L^T  i1:i-i j1:j0-j0 */
	for (i = 1; i <= n1; i++)
	{
		if (n[i][2] == 1)   goto c006;
		for (j = n[i][1]; j <= n[i][1] + n[i][2] - 1; j++)

/**************************************************/
/* df21: backward and forward substitution        */
/**************************************************/
void df21()
{
	if (k6 == 1)  fprintf(fl0, "---21000--- \n");
	/*------------------------------------------------*/
	/* F_i - S_k=1^i-1 L_ik Y_k  i:i k:i1=k[j]   back */
	for (i = 1; i <= n1; i++)
	{
		for (j = n[i][1]; j <= n[i][1] + n[i][2] - 2; j++)
		{
			i1 = k[j];
			for (l = 1; l <= 6; l++)
			{
				for (l1 = 1; l1 <= 6; l1++)
				{
					f[i][l] += -a[j][6 * (l - 1) + l1] * f[i1][l1];
				}  /*  l1 */
			}  /*  l  */
		}  /*  j  */

		/*------------------------------------------------*/
		/* Y_i = L_ii^-1 (F_i - S_k=1^i-1 L_ik Y_k)  back */
		for (j = 1; j <= 6; j++)
		{
			s[j] = 0;
			for (l = 1; l <= 6; l++)
			{
				s[j] += v[i][6 * (j - 1) + l] * f[i][l];
			}  /*  l  */
		}  /*  j  */
		/*------------------------------------------------*/
		for (j = 1; j <= 6; j++)
		{
			f[i][j] = s[j];
		}  /*  j  */
	}  /*  i  */
	/**************************************************/
	/* find L_ki^T  k>i  i:i k:j=>j1             fore */
	for (i = n1 - 1; i >= 1; i--)
	{
		for (j = i + 1; j <= n1; j++)
		{
			for (l = n[j][1]; l <= n[j][1] + n[j][2] - 1; l++)
			{
				ji = l;
				if (k[l] == i) goto c101;
			}  /*  l  */
			goto c102;
		c101:;
			/*------------------------------------------------*/
			/* Q=L_ii^(-1)  E=L_ki^T  E => E^T          fore  */
			for (l = 1; l <= 6; l++)
			{
				for (l1 = 1; l1 <= 6; l1++)
				{
					q[l][l1] = v[i][6 * (l - 1) + l1];
					e[l1][l] = a[ji][6 * (l - 1) + l1];
				}  /*  l1 */
			}  /*  l  */
			/*------------------------------------------------*/
			/* E=L_ki^T  F[j]=X_k  E*F => T=L_ki^T X_k   fore */
			for (l = 1; l <= 6; l++)
			{
				t[l][1] = 0;
				for (l1 = 1; l1 <= 6; l1++)
				{
					t[l][1] += e[l][l1] * f[j][l1];
				}  /*  l1 */
			}  /*  l  */
			/*------------------------------------------------*/
			/* X_i = Y_i - S_(k=i+1)^n L_ii^(-1) L_ki^T X_k   */
			for (l = 1; l <= 6; l++)
			{
				for (l1 = 1; l1 <= 6; l1++)
				{
					f[i][l] += -q[l][l1] * t[l1][1];
				}  /*  l1 */
			}  /*  l  */
		c102:;
		}  /*  j  */
	}  /*  i  */
	/*------------------------------------------------*/
	for (i = 1; i <= n1; i++)
	{
		for (j = 1; j <= 6; j++)
		{
			z[i][j] = f[i][j];
		}  /*  j  */
	}  /*  i  */

		a3 = h0[i][1] * e0[i][0] * sqrt(a1 * a1 + a2 * a2);
		o1 += a3;
		if (n[i][0] == 1) o2 += a3;
	}  /*  i  */
	o1 /= o0;
	fprintf(fl0, "=====unit inert %lf ===== \n", o1);
	/* printf("unit inert %lf  \n",o1);               */
	/**************************************************/
	/* factor of safty ff                             */
	/* o2: sliding force          o0: resistant force */
	o0 = 0;
	o2 += .0000000001;
	for (i = 1; i <= n2; i++)
	{
		l1 = m[i][1];
		l2 = m[i][2];
		i1 = n0[l1];
		i2 = n0[l2];
		if (n[i1][0] == 0 && n[i2][0] == 0) goto c503;
		if (n[i1][0] == 1 && n[i2][0] == 1) goto c503;
		/*------------------------------------------------*/
		i0 = m1[i][0];
		t1 = b0[i0][0];
		t2 = b0[i0][1] * o[i][5];
		/*------------------------------------------------*/
		if (m[i][0] == 1) goto c503;
		if (m0[i][2] == 0) goto c503;
		if (m0[i][2] == 1) goto c502;
		if (m0[i][2] == 2) goto c501;
		/*------------------------------------------------*/
		/* normal spring and shear spring                 */
	c501:;
		e1 = 0;
		if (m0[i][0] == 2)  e1 = t2;
		a2 = tan(dd * t1);
		if (o[i][3] > 0)  a2 = 0;
		o0 += fabs(o[i][3]) * a2 * hh + e1;
		o2 += fabs(o[i][4]) * hh / h2;
		goto c503;
		/*------------------------------------------------*/
		/* normal spring and friction force               */
	c502:;
		a2 = tan(dd * t1);
		if (o[i][3] > 0)  a2 = 0;
		o0 += fabs(o[i][3]) * a2 * hh;
		o2 += fabs(o[i][3]) * a2 * hh;
	c503:;
	}  /*  i  */
	ff = o0 / o2;
	fprintf(fl0, "fs= %lf  \n", ff);
	/* printf("fs= %lf  \n",ff);                     */
	/*------------------------------------------------*/
	/* compute s sx sy sxx syy sxy    g0 of this step */
	area();
	/*------------------------------------------------*/
}

/**************************************************/
/* dspl: block displacement matrix                */
/**************************************************/
/* x10  y10  u2  v2  t[][]               i0  x  y */
void dspl()
{
	/*------------------------------------------------*/
	/* i0 is old block number                         */
	/* x10,y10:center of gravity of the block         */
	x10 = h0[i0][2] / h0[i0][1];
	y10 = h0[i0][3] / h0[i0][1]; 
	/*------------------------------------------------*/
	u2 = x - x10; 
	v2 = y - y10;  
	t[1][1] = 1;
	t[2][1] = 0;
	t[1][2] = 0;
	t[2][2] = 1;
	t[1][3] = -v2;
	t[2][3] = u2;
	t[1][4] = u2;
	t[2][4] = 0;
	t[1][5] = 0;
	t[2][5] = v2;
	t[1][6] = v2 / 2;
	t[2][6] = u2 / 2;
	/*------------------------------------------------*/
}

/**************************************************/
/* invr: inverse of 6*6 matrix                    */
/**************************************************/
/* jj  j  l  l1  e[][]  q[][]                     */
void invr()
{
	/*------------------------------------------------*/
	/* E=I as n=6 free term colomns                   */
	for (j = 1; j <= 6; j++)
	{
		for (l = 1; l <= 6; l++)
		{
			e[j][l] = 0; 
		}  /*  l  */
		e[j][j] = 1;  
	}  /*  j  */
	/*------------------------------------------------*/
	/* L_ij = A_ij - S_(k=1)^j-1 L_ik L_kk^-1 L_jk^T  */
	for (j = 1; j <= 6; j++)
	{
		for (l = 1; l <= j; l++)
		{
			for (l1 = 1; l1 <= l - 1; l1++)
			{
				q[j][l] += -q[j][l1] * q[l][l1] / q[l1][l1];  
			}  /*  l1 */
		}  /*  l  */
	}  /*  j  */
	/*------------------------------------------------*/
	/* Y_i = L_ii^-1 (F_i - S_k=1^i-1 L_ik Y_k)  back */
	for (jj = 1; jj <= 6; jj++)
	{
		for (j = 1; j <= 6; j++)
		{
			for (l = 1; l <= j - 1; l++)
			{
				e[j][jj] += -q[j][l] * e[l][jj];
			}  /*  l  */
			e[j][jj] /= q[j][j]; 
		}  /*  j  */
	}  /*  jj */
	/*------------------------------------------------*/
	/* X_i = Y_i - S_(k=i+1)^n L_ii^(-1) L_ki^T X_k   */
	for (jj = 1; jj <= 6; jj++)
	{
		for (j = 5; j >= 1; j--)
		{
			for (l = j + 1; l <= 6; l++)
			{
				e[j][jj] += -q[l][j] * e[l][jj] / q[j][j]; 
			}  /*  l  */
		}  /*  j  */
	}  /*  jj */
	/*------------------------------------------------*/
}

/**************************************************/
/* mult: muliplication of 6*6 matrices            */
/**************************************************/
/* l1  l2  l3  t[][]  q[][]                       */
void mult()
{
	/*------------------------------------------------*/
	for (l1 = 1; l1 <= 6; l1++)
	{
		for (l2 = 1; l2 <= 6; l2++)
		{
			t[l1][l2] = 0;
			for (l3 = 1; l3 <= 6; l3++)
			{
				t[l1][l2] += q[l1][l3] * e[l3][l2];  
			}  /*  l3 */
		}  /*  l2 */
	}  /*  l1 */
	/*------------------------------------------------*/
	for (l1 = 1; l1 <= 6; l1++)
	{
		for (l2 = 1; l2 <= 6; l2++)
		{
			q[l1][l2] = t[l1][l2];  
		}  /*  l2 */
	}  /*  l1 */
	/*------------------------------------------------*/
}

/**************************************************/
/* cons: set control constants                    */
/**************************************************/
void cons()
{
	/*------------------------------------------------*/
	/* pre-defined changeble constants                */
	/* f0 : criteria of open to close                 */
	/* h1 : angle overlapping degrees     for contact */
	/* h2 : ratio for shear spring versus normal one  */
	/* h3 : coefficient of spring dspl to d0   dstn() */
	/* h4 : coefficient of spring stiffness  g0=h4*em */
	/* h5 : coefficient of spring invasion  d0 df04() */
	h1 = 3;   //
	h2 = 2.5; //h2*ks=kn
	h3 = 1.1;
	h4 = 40.0;//
	h5 = 0.4;
	/*------------------------------------------------*/
	/* d0 :  normal external    distance for contact  */
	/* d0*.4 normal penetration distance for contact  */
	/* this is unsed alternative define d0 by w1 w2   */
	d9 = 2.5 * w0 * g2;
	a1 = w2 / h5;
	if (a1 > w1)  a1 = w1;
	a1 /= 7.50;
	/* fprintf(fl0,"### d0-in: %lf d0-ge %lf ### \n",d9,a1);
	fprintf(fl0,"### d0-in: %lf d0-ge %lf ### \n",d9,a1); */
	if (d9 < a1) d9 = a1; 
	d0 = d9;
	/*------------------------------------------------*/
	/* define d0 from fixed rate f2 of w0             */
	/* g2 d0 small penetration h5d0 & d0 same h5 .3-1 */
	f2 = .0004;
	d0 = 2.5 * w0 * g2;
	h5 = 0.3;
	a1 = 0.3 * d0;
	if (a1 < 3.0 * f2 * w0) { a1 = 3.0 * f2 * w0; h5 = a1 / d0; }
	if (d0 < a1) { d0 = a1;        h5 = 1.0; }
	/*------------------------------------------------*/
co01:;
	hh = g0;

	hh1 = hh / averEdgeLen;  

	a1 = 0;
	for (i = 1; i <= nb; i++)
	{
		a1 += a0[i][3];
	}  /*  i  */
	a1 /= nb;
	if (g0 < .00000000001)  hh = h4 * a1;
	g6 = hh;
	g7 = hh;
	/*------------------------------------------------*/
	f0 = .0000001;                             /* ??? */
	/*------------------------------------------------*/
}

/**************************************************/
/* step: compute next time interval               */
/**************************************************/
/* compute tt                                     */
void step()
{
	double L0, L1, L01;
	/*------------------------------------------------*/
	if (g1 < .00000000001) goto st03;
	/* if (nn ==                 1) */
	tt = g1;
	if (nn > 1 && q0[nn - 1][0] > 1.1)  tt /= sqrt(q0[nn - 1][0]);
	goto st02;
	/*------------------------------------------------*/
	/* body load       a1=g: (unit force)/(unit mass) */
st03:;
	if (nn > 1) goto st01;
	a1 = 0;
	for (i = 1; i <= nb; i++)
	{
		b1 = sqrt(a0[i][1] * a0[i][1] + a0[i][2] * a0[i][2]) / a0[i][0];
		if (a1 < b1)  a1 = b1;
	}  /*  i  */
	/*------------------------------------------------*/
	/* point load per unit mass                       */
	a2 = 0;
	for (i = nf + 1; i <= nf + nl; i++)
	{
		for (j = k4[i][0]; j <= k4[i][1]; j++)
		{
			b1 = sqrt(u0[j][1] * u0[j][1] + u0[j][2] * u0[j][2]) / (w3 * a0[1][0]);
			if (a2 < b1)  a2 = b1;
		}  /*  j  */
	}  /*  i  */
	/*------------------------------------------------*/
	/* initial stresses as unit mass body force  a1=a */
	a3 = 0;
	for (i = 1; i <= nb; i++)
	{
		for (j = 5; j <= 7; j++)
		{
			b1 = fabs(a0[i][j]) / a0[i][0];
			if (a3 < b1) a3 = b1;
		}  /*  j  */
	}  /*  i  */
	a3 = 4 * a3 * sqrt(w3) / w3;
	a1 = (a1 + a2 + a3) / 2;
	/*------------------------------------------------*/
	/* initial velocity                  a2: velocity */
	a2 = 0;
	for (i = 1; i <= nb; i++)
	{
		b1 = sqrt(a0[i][11] * a0[i][11] + a0[i][12] * a0[i][12] + a0[i][13] * a0[i][13]);
		if (a2 < b1)  a2 = b1;
	}  /*  i  */
	/*------------------------------------------------*/
	/* s=1/2*g*t*t             a1*t*t + a2*t + a3 = 0 */
	a3 = -g2 * w0;
	tt = (-a2 + sqrt(a2 * a2 - 4 * a1 * a3)) / (2 * a1);
	goto st02;
	/**************************************************/
	/* body load       a1=g: (unit force)/(unit mass) */
st01:;
	a3 = 0;
	for (i = nf + 1; i <= nf + nl; i++)
	{
		if (a3 < fabs(g[i][4])) a3 = fabs(g[i][4]);
		if (a3 < fabs(g[i][5])) a3 = fabs(g[i][5]);
	}  /*  i  */
	a3 = a3 / (w3 * a0[1][0]);
	a1 = w7 * hh / (w3 * a0[1][0]);
	if (a1 < a3)  a1 = a3;
	if (a1 == 0)  goto st02;
	/*------------------------------------------------*/
	a2 = q0[nn - 1][0] * g2 * w0 / tt;
	if (gg < 0.1) a2 = 0;
	a3 = -g2 * w0;
	a9 = (-a2 + sqrt(a2 * a2 - 2 * a1 * a3)) / a1;

	L0 = 10.0;  L1 = 10.0;
	L01 = a1 * tt / (4 * a2) * (1.0 / L0 / L0 - 1) / L1 + 1;
	if (a1 * tt / (4 * a2) > 10)
		L01 = 1.0 / L0 / L0;

	a1 = a1 * L01;
	a9 = (-a2 + sqrt(a2 * a2 - 2 * a1 * a3)) / a1; 



	if (a9 < 1.3 * tt) tt = a9;
	if (a9 >= 1.3 * tt) tt = 1.3 * tt;
	/*------------------------------------------------*/
st02:;
}

/**************************************************/
/* dstn: compute stiffness of contact spring      */
/**************************************************/
void dstn()
{
	/*------------------------------------------------*/
	/* w6 max penetration                             */
	w6 = 0;
	w7 = 0;
	for (i = 1; i <= n2; i++)
	{
		if (m0[i][2] == 0) goto ds01;
		/*------------------------------------------------*/
		b1 = o[i][3];
		if (fabs(w6) < fabs(b1))  i2 = i;
		if (fabs(w6) < fabs(b1))  w6 = b1;
		if (w7 > o[i][3])  w7 = o[i][3];
	ds01:;
	}  /*  i  */
	w7 = -w7;
	/* fprintf(fl0,"wf/d0 %lf i %d \n",w6/d0,i2); */
	/*------------------------------------------------*/
	/* updating g0 hh by max penetration w6           */
	if (g0 > .00000000001) goto ds02;
	b1 = hh * fabs(w6 / w0) / f2 + 0.0000001;
	if (b1 <= hh / 3.0)  b1 = hh / 3.0;
	hh = b1;
	/**************************************************/
	/* in step penetration control iteration of w6    */
	/* transfer o3 to o0 o1         transter o4 to o1 */
ds02:;
	for (i = 1; i <= n2; i++)
	{
		if (m0[i][2] == 0) o[i][0] = 0;
		if (m0[i][2] == 0) o[i][1] = 0;
		if (m0[i][2] == 1) o[i][0] = o[i][3];
		if (m0[i][2] == 1) o[i][1] = 0;
		if (m0[i][2] == 3) o[i][1] = o[i][3];
		if (m0[i][2] == 3) o[i][0] = 0;
		if (m0[i][2] == 2) o[i][0] = o[i][3];
		if (m0[i][2] == 2) o[i][1] = o[i][4];
	}  /*  i  */
}

}

/**************************************************/
/* area: compute s0 sx sy sxx syy sxy             */
/**************************************************/
/* i  j  x2  y2  x3  y3  h0[][]                   */
void area()
{
	/*------------------------------------------------*/
	for (i = 1; i <= n1; i++)
	{
		for (j = 1; j <= 6; j++)
		{
			h0[i][j] = 0;
		}  /*  j  */
		/*------------------------------------------------*/
		/* x1=0  y1=0 here                                */
		for (j = k0[i][1]; j <= k0[i][2]; j++)
		{
			x2 = d[j][1];
			y2 = d[j][2];
			x3 = d[j + 1][1];
			y3 = d[j + 1][2];
			/*------------------------------------------------*/
			s0 = (x2 * y3 - x3 * y2);    
			h0[i][1] += s0 / 2;             
			h0[i][2] += s0 * (0 + x2 + x3) / 6;   
			h0[i][3] += s0 * (0 + y2 + y3) / 6;  
			h0[i][4] += s0 * (x2 * x2 + x3 * x3 + x2 * x3) / 12; 
			h0[i][5] += s0 * (y2 * y2 + y3 * y3 + y2 * y3) / 12; 
			h0[i][6] += s0 * (2 * x2 * y2 + 2 * x3 * y3 + x2 * y3 + x3 * y2) / 24; 
		}  /*  j  */
	}  /*  i  */
	/*------------------------------------------------*/
	for (i = 1; i <= n1; i++)
	{
		if (h0[i][1] < .0000000000000001)
			fprintf(fl0, "minus area block=%d area=%lf \n", i, h0[i][1]);
	}  /*  i  */
	/*------------------------------------------------*/
}

/**************************************************/
/* angl: material constants of bolts              */
/**************************************************/
/* i  x1  y1  c1  d1  d2  d3  i1  i2  u[][]       */
void angl()
{
	/*------------------------------------------------*/
	for (i = 1; i <= n1; i++)
	{
		i1 = k0[i][1];
		i2 = k0[i][2];
		/*------------------------------------------------*/
		/* u[j][1]: direction angle of edge j j+1         */  
		for (j = i1 - 1; j <= i2; j++)
		{
			x1 = d[j + 1][1] - d[j][1];
			yi = d[j + 1][2] - d[j][2];
			c1 = fabs(x1) + .0000001;
			d1 = atan(yi / c1) / dd;
			if (x1 < 0)  d1 = 180 - d1;
			if (d1 < 0)  d1 += 360;
			u[j][1] = d1;
		}  /*  j  */
		u[i2 + 1][1] = u[i1][1];      
		/*------------------------------------------------*/
		/* u[j][2]: angle j                               */  
		for (j = i1; j <= i2; j++)
		{
			d1 = u[j][1];
			d2 = u[j - 1][1] - 180;
			if (d2 < 0)  d2 += 360;
			d3 = d2 - d1;
			if (d3 < 0)  d3 += 360;
			u[j][2] = d3;
		}  /*  j  */
		u[i1 - 1][2] = u[i2][2];
		u[i2 + 1][2] = u[i1][2];
	}  /*  i  */
	/*------------------------------------------------*/
}


void dist()
{
	double x4, y4;
	double t1;
	t1 = (x1 - x2) * (x3 - x2) + (yi - y2) * (y3 - y2);
	t1 /= ((x3 - x2) * (x3 - x2) + (y3 - y2) * (y3 - y2));
	if (t1 <= 0.0)  t1 = 0;
	if (t1 >= 1.0)  t1 = 1;
	x4 = x2 + t1 * (x3 - x2);
	y4 = y2 + t1 * (y3 - y2);   
	d5 = sqrt((x4 - x1) * (x4 - x1) + (y4 - yi) * (y4 - yi));
	/*------------------------------------------------*/
}

/**************************************************/
/* proj: projection of a edge to other edge       */
/**************************************************/
/* i  i0   i1  i2  i3  i4  d1  o[][5]  d  m0[][2] */
/* j l d1 i0 s[] x1 y1 x2 y2 x3 y3                */
void proj()
{
	void movl();
	void dral();
	/*------------------------------------------------*/
	/* compute projection of p1 and p4 to edge p2p3   */
	x1 = d[i1][1];
	yi = d[i1][2]; 
	x2 = d[i2][1];
	y2 = d[i2][2];
	x3 = d[i3][1];
	y3 = d[i3][2]; 
	/*------------------------------------------------*/
	/* contact point o[][2]  s[1-4] par p1-p4 to p2p3 */
	/* k[i]=1 transfer contact point m0[][0]=1 record */
	d1 = ((x3 - x2) * (x3 - x2) + (y3 - y2) * (y3 - y2) + .000000001 * w0); 
	s[1] = ((x1 - x2) * (x3 - x2) + (yi - y2) * (y3 - y2)) / d1;   
	if (k[i] == 0)  o[i][2] = s[1]; 
	if (k[i] == 1 && m0[i][0] == 0)  m0[i][0] = 1; 
	s[2] = 0;
	s[3] = 1;
	/*------------------------------------------------*/
	if (i4 == 0) goto pr02;
	x1 = d[i4][1];
	yi = d[i4][2];
	s[4] = ((x1 - x2) * (x3 - x2) + (yi - y2) * (y3 - y2)) / d1; 
	/*------------------------------------------------*/
	/* ordering p1 p2 p3 p4 on edge p2p3              */
	for (j = 1; j <= 3; j++)
	{
		for (l = j + 1; l <= 4; l++)
		{
			if (s[j] <= s[l]) goto pr01;
			s[0] = s[j];
			s[j] = s[l];
			s[l] = s[0]; 
		pr01:;
		}  /*  l  */
	}  /*  j  */
	/*------------------------------------------------*/
	/* o[][5] 1/2 co-edge lock length                 */
	o[i][5] += .5 * (s[3] - s[2]) * sqrt(d1); 
	/**************************************************/
	/* draw co-edge or lock tension length            */
	if (m0[i][0] != 2)  goto pr02;
	s[5] = .5 * (s[2] + s[3]);  
	pe = 3;
	x8 = d[i1][1];
	y8 = d[i1][2]; 
	movl(x8, y8, pe, p0, w);
	x9 = (1 - s[5]) * d[i2][1] + s[5] * d[i3][1];
	y9 = (1 - s[5]) * d[i2][2] + s[5] * d[i3][2];
	dral(x9, y9, pe, p0, w);
	/*------------------------------------------------*/
pr02:;
}

		/* EndPaint(hwnd, &paintstruct);                  */
		break;
		/**************************************************/
		/* terminate the program              delete pens */
	case WM_DESTROY:
	case WM_CLOSE:
		DeleteObject(hbrush[1]);
		DeleteObject(hbrush[2]);
		DeleteObject(hbrush[3]);
		DeleteObject(hbrush[4]);
		DeleteObject(hbrush[5]);
		DeleteObject(hbrush[6]);
		DeleteObject(hbrush[7]);
		DeleteObject(hbrush[8]);
		DeleteObject(hbrush[9]);
		DeleteObject(hbrush[10]);
		DeleteObject(hOldbrush);
		/*------------------------------------------------*/
		DeleteObject(hpen[1]);
		DeleteObject(hpen[2]);
		DeleteObject(hpen[3]);
		DeleteObject(hpen[4]);
		DeleteObject(hpen[5]);
		DeleteObject(hOldpen);
		/*------------------------------------------------*/
		DeleteObject(hnewf1);
		DeleteObject(hnewf2);
		DeleteDC(memdc);
		PostQuitMessage(0);
		break;
		/**************************************************/
		/* any message not specified in preceding switch  */
	default:
		return DefWindowProc(hwnd, message, wParam, lParam);
	}  /* switch */
	return 0;
}

	/* df04: contact finding by distance criteria     */
	/* df05: contact finding by angle criteria        */
	/* df06: contact transfer                         */
	/* df07: contact initialization                   */
	/* df08: positions of non-zero storage            */
	/* df08: block order                              */
	/* df09: time interpolation                       */
	/*------------------------------------------------*/
	/* df10: initiate coefficient and load matrix     */
	/* df11: submatrix of inertia                     */
	/* df12: submatrix of fixed points                */
	/* df13: submatrix of stifness                    */
	/* df14: submatrix of initial stress              */
	/* df15: submatrix of point loading               */
	/* df16: submatrix of volume force                */
	/*------------------------------------------------*/
	/* df18: iteration start                          */
	/* df18: add and subtract submatrix of contact    */
	/* df20: triangle discomposition equation solver  */
	/* df21: backward and forward substitution        */
	/* df22: contact judge after iteration            */
	/* df22: iteration output                         */
	/* df24: displacement ratio and iteration drawing */

	/* df27: save results to file                     */
	/*------------------------------------------------*/
	/* dspl: block displacement matrix                */
	/* invr: inverse of 6*6 matrix                    */
	/* mult: muliplication of 6*6 matrices            */
	/* sign: return sign                              */
	/* cons: set control constants                    */

	/* exti: extent 2-d integer arrays                */
	/* extd: extent 2-d double  arrays                */
	/* area: compute s0 sx sy sxx syy sxy             */

	/* mjoi: material constants of joints             */
	/* angl: material constants of bolts              */
	clock_t start, finish;

	start = clock();

	wcl.hInstance = hThisInst;

	hdc = BeginPaint(hwnd, &paintstruct);

	nn = 0;

	df01();
	df02();

	df26();
	/*------------------------------------------------*/
	/* df03 set step time interval  contact stiffness */
	/* pp=0 direct     variable g0 by penetration  f2 */
	/* pp=1 iteration  variable g0 by penetration  f2 */

	for (nn = 1; nn <= n5; nn++)
	{
		for (mi = 1; mi <= n2; mi++) o[mi][7] = o[mi][6];

		df04();
		df05();

		for(i=1;i<=n1; i++)
